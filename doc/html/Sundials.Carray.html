Module Sundials.Carray


module Carray: sig .. end
Utility functions for serial nvectors as used in Cvode_serial.
type t = Sundials.real_array 
val empty : t
An array with 0 elements.
val create : int -> t
create n returns an array with n elements.
val of_array : float array -> t
Copies the contents of an Array into a Sundials.real_array
val of_carray : t -> t
Create a new array with the same contents as an existing one.
val blit : t -> t -> unit
fill src dst copies all elements of array src into array dst. They must both have the same length.
val fill : t -> float -> unit
fill a c sets all elements of the array a to the constant c.
val length : t -> int
Returns the length of an array
val print_with_time : float -> t -> unit
print_with_time t a prints a line containing the current time (see Sundials.print_time) followed by a tab-delimited list of the values of a, using the format % e, and then a newline.
val print_with_time' : float -> t -> unit
print_with_time' t a prints a line containing the current time (see Sundials.print_time) followed by a tab-delimited list of the values of a, using the format % .8f, and then a newline.
val print_with_time'' : float -> t -> unit
print_with_time'' t a prints a line containing the current time (see Sundials.print_time) followed by a tab-delimited list of the values of a, using the format % .15f, and then a newline.
val app : (float -> unit) -> t -> unit
app f a applies f to the values of each element in a.
val appi : (int -> float -> unit) -> t -> unit
app f a applies f to the indexes and values of each element in a.
val map : (float -> float) -> t -> unit
map f a applies f to the value of each element in a and stores the result back into the same element.
val mapi : (int -> float -> float) -> t -> unit
map f a applies f to the index and value of each element in a and stores the result back into the same element.
